*/Operações com Matrizes 2x4 usando Funções

A última atividade pede a criação de funções que recebam três matrizes (A, B e C, todas 2x4) e coloquem os resultados de certas operações matemáticas diretamente na matriz C. As operações solicitadas são a subtração A−B e a multiplicação por um escalar 3×A
/*
#include <stdio.h>

// Função a) Subtração A - B guardada na matriz C
void subtraiMatrizes(int A[2][4], int B[2][4], int C[2][4]) {
    for(int i = 0; i < 2; i++) {
        for(int j = 0; j < 4; j++) {
            C[i][j] = A[i][j] - B[i][j];
        }
    }
}

// Função b) Multiplicação 3 x A guardada na matriz C
void multiplicaMatriz(int A[2][4], int C[2][4]) {
    for(int i = 0; i < 2; i++) {
        for(int j = 0; j < 4; j++) {
            C[i][j] = 3 * A[i][j];
        }
    }
}

// Função auxiliar para imprimir a matriz (aplicando o que foi ensinado no slide 3)
void imprimeMatriz2x4(int matriz[2][4]) {
    for(int i = 0; i < 2; i++) {
        printf("| ");
        for(int j = 0; j < 4; j++) {
            printf("%4d ", matriz[i][j]);
        }
        printf("|\n");
    }
}

int main() {
    int matA[2][4] = {{5, 5, 5, 5}, {10, 10, 10, 10}};
    int matB[2][4] = {{1, 2, 3, 4}, {1, 2, 3, 4}};
    int matC[2][4];

    printf("Operacao A - B:\n");
    subtraiMatrizes(matA, matB, matC);
    imprimeMatriz2x4(matC);

    printf("\nOperacao 3 x A:\n");
    multiplicaMatriz(matA, matC);
    imprimeMatriz2x4(matC);

    return 0;
}
