// Restrição rígida: tamanho máximo para transferência na RAM
#define BUFFER_SIZE 4096 

// 1. Estrutura do Superbloco (fica no byte 0 do arquivo .pak)
typedef struct {
    int total_assets;
    long manifest_offset; // Onde no arquivo começa a tabela de metadados?
} PakHeader;

// 2. Estrutura do Metadado (compõe o diretório/manifesto no final do arquivo)
typedef struct {
    char asset_name[256];
    size_t file_size;
    time_t pack_date;
    long byte_offset_in_pak;
    int is_deleted; // Flag para remoção lógica (0 = ativo, 1 = removido)
} AssetEntry;

// Variáveis globais em memória (conforme restrições)
AssetEntry *manifest = NULL; // Alocado dinamicamente com malloc/realloc
PakHeader header = {0, sizeof(PakHeader)}; // Inicializa apontando o manifesto logo após o header

// Função de comparação para o qsort (ordenação de arrays em C)
int compare_by_size(const void *a, const void *b) {
    AssetEntry *assetA = (AssetEntry *)a;
    AssetEntry *assetB = (AssetEntry *)b;
    // Retorna a diferença para ordenar de forma crescente
    return (assetA->file_size - assetB->file_size); 
}
